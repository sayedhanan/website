  java={`import java.util.*;

public class SimpleChatbot {
    private Map<String, String[]> responses;
    private Random random;
    
    public SimpleChatbot() {
        this.random = new Random();
        this.responses = new HashMap<>();
        
        responses.put("hello", new String[]{"Hi there!", "Hello!", "Hey!"});
        responses.put("how are you", new String[]{"I'm good, thanks!", "Doing well!", "Great!"});
        responses.put("bye", new String[]{"Goodbye!", "See you later!", "Bye!"});
    }
    
    public String getResponse(String userInput) {
        userInput = userInput.toLowerCase().trim();
        
        for (String key : responses.keySet()) {
            if (userInput.contains(key)) {
                String[] responseArray = responses.get(key);
                return responseArray[random.nextInt(responseArray.length)];
            }
        }
        
        return "I don't understand that.";
    }
    
    // Usage
    public static void main(String[] args) {
        SimpleChatbot bot = new SimpleChatbot();
        String response = bot.getResponse("hello");
        System.out.println(response);
    }
}`}
